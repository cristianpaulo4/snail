Future<int> deleteMany(List<ID> ids) async {
  final db = await _getDatabase();
  var result = await db.delete(
    tableName,
    where:
        '$primaryKeyColumn IN (${List.filled(ids.length, '?').join(', ')})',
    whereArgs: ids,
  );
  return result;
}
